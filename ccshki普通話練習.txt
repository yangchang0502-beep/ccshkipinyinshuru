                const osc = this.ctx.createOscillator();

        function getInitials(pinyin) {
            const normalized = normalizePinyin(pinyin).toLowerCase();
            const initials = ['zh', 'ch', 'sh', 'b', 'p', 'm', 'f', 'd', 't', 'n', 'l', 'g', 'k', 'h', 'j', 'q', 'x', 'r', 'z', 'c', 's', 'y', 'w'];
            for (const initial of initials) { if (normalized.startsWith(initial)) return initial; }
            return '';
        }

        function getFinals(pinyin) {
            const normalized = normalizePinyin(pinyin).toLowerCase();
            const initial = getInitials(normalized);
            return initial ? normalized.slice(initial.length) : normalized;
        }

        function convertNumberToneToSymbol(pinyin) {
            const match = pinyin.match(/^([a-zÃ¼A-ZÃœÄ€Ä’ÄªÅŒÅªÇ•ÃÃ‰ÃÃ“ÃšÇ—ÇÄšÇÇ‘Ç“Ç™Ã€ÃˆÃŒÃ’Ã™Ç›]+)([1-5]?)$/);
            if (!match) return pinyin;
            let [, syllable, tone] = match;
            tone = parseInt(tone) || 5;
            if (tone === 5) return syllable;
            syllable = syllable.toLowerCase();
            let result = syllable;
            if (syllable.includes('a')) result = syllable.replace('a', toneMarks['a'][tone]);
            else if (syllable.includes('o')) result = syllable.replace('o', toneMarks['o'][tone]);
            else if (syllable.includes('e')) result = syllable.replace('e', toneMarks['e'][tone]);
            else if (syllable.includes('iu')) result = syllable.replace('u', toneMarks['u'][tone]);
            else if (syllable.includes('ui')) result = syllable.replace('i', toneMarks['i'][tone]);
            else if (syllable.includes('i')) result = syllable.replace('i', toneMarks['i'][tone]);
            else if (syllable.includes('u')) result = syllable.replace('u', toneMarks['u'][tone]);
            else if (syllable.includes('Ã¼') || syllable.includes('v')) result = syllable.replace(/[Ã¼v]/, toneMarks['Ã¼'][tone]);
            return result;
        }

        function convertSymbolToneToNumber(pinyin) {
            const tonePattern = /[ÄÄ“Ä«ÅÅ«Ç–Ã¡Ã©Ã­Ã³ÃºÇ˜ÇÄ›ÇÇ’Ç”ÇšÃ Ã¨Ã¬Ã²Ã¹Çœ]/;
            const match = pinyin.match(tonePattern);
            if (!match) return pinyin + '5';
            const char = match[0];
            let tone = 5;
            for (const [vowel, marks] of Object.entries(toneMarks)) {
                const index = marks.indexOf(char);
                if (index > 0) { tone = index; break; }
            }
            let result = pinyin;
            for (const [vowel, marks] of Object.entries(toneMarks)) {
                for (let i = 1; i < marks.length; i++) result = result.replace(marks[i], marks[0]);
            }
            return result + tone;
        }

        function normalizePinyin(pinyin) {
            let result = pinyin.toLowerCase();
            for (const [vowel, marks] of Object.entries(toneMarks)) {
                for (let i = 1; i < marks.length; i++) {
                    result = result.replace(new RegExp(marks[i], 'g'), marks[0]);
                }
            }
            result = result.replace(/v/g, 'Ã¼').replace(/[1-5]$/, '');
            return result;
        }

        function comparePinyin(userInput, correctAnswer) {
            userInput = userInput.trim().replace(/\s+/g, ' ');
            correctAnswer = correctAnswer.trim().replace(/\s+/g, ' ');
            if (userInput.toLowerCase() === correctAnswer.toLowerCase()) return true;
            if (correctAnswer.includes(' ')) return compareMultiSyllablePinyin(userInput, correctAnswer);
            return compareSingleSyllablePinyin(userInput, correctAnswer);
        }

        function compareMultiSyllablePinyin(userInput, correctAnswer) {
            const userSyllables = userInput.split(/\s+/);
            const correctSyllables = correctAnswer.split(/\s+/);
            if (userSyllables.length !== correctSyllables.length) return false;
            for (let i = 0; i < userSyllables.length; i++) {
                if (!compareSingleSyllablePinyin(userSyllables[i], correctSyllables[i])) return false;
            }
            return true;
        }

        function compareSingleSyllablePinyin(userInput, correctAnswer) {
            if (userInput.toLowerCase() === correctAnswer.toLowerCase()) return true;
            const userWithNumber = userInput.match(/\d$/) ? userInput : convertSymbolToneToNumber(userInput);
            const correctWithNumber = correctAnswer.match(/\d$/) ? correctAnswer : convertSymbolToneToNumber(correctAnswer);
            return userWithNumber.toLowerCase() === correctWithNumber.toLowerCase();
        }

            if (!value) return;
            const btn = document.querySelector(`.tone-${tone}`);
            btn.classList.remove('animate-jelly'); void btn.offsetWidth; btn.classList.add('animate-jelly');
            if (value.includes(' ')) {
                const syllables = value.split(' ');
                const lastSyllableIndex = syllables.length - 1;
                let lastSyllable = normalizePinyin(syllables[lastSyllableIndex]);
                syllables[lastSyllableIndex] = tone === 5 ? lastSyllable : convertNumberToneToSymbol(lastSyllable + tone);
                input.value = syllables.join(' ');
            } else {
                const cleanSyllable = normalizePinyin(value);
                input.value = tone === 5 ? cleanSyllable : convertNumberToneToSymbol(cleanSyllable + tone);
            }
            input.focus();
        }

        function toggleHint(type) {
            const question = questions[currentQuestion];
            if (!question) return;
            const hintDisplay = document.getElementById('hintDisplay');
            let hintText = '';
            if (question.isArticleSentence) {
                const currentCharData = question.displayChars[question.currentCharIndex];
                if (currentCharData && !currentCharData.isPunctuation) {
                    const currentPinyin = currentCharData.pinyin;
                    switch(type) {
                        case 'initials': hintText = getInitials(currentPinyin); break;
                        case 'finals': hintText = getFinals(currentPinyin); break;
                        case 'full': hintText = currentPinyin; break;
                    }
                }
            } else if (question.isWord && question.pinyinArray) {
                switch(type) {
                    case 'initials': hintText = question.pinyinArray.map(p => getInitials(p)).join(' '); break;
                    case 'finals': hintText = question.pinyinArray.map(p => getFinals(p)).join(' '); break;
                    case 'full': hintText = question.pinyin; break;
                }
            }
            if (hintDisplay.textContent === hintText) hintDisplay.textContent = '';
            else { hintDisplay.textContent = hintText; hintUsed = true; }
        }

            }, 1000);
        }
